using namespace std;

typedef pair < int, int > pii;
typedef pair < ll, ll > pll;

ll d[105][100005];

struct Stone {
	int s, e, l;
	bool operator < (const Stone &a) {
		return l * a.s > a.l * s;
	}
}a[105];


int main()
{
    ios::sync_with_stdio(0);
    cin.tie(NULL);

	int t;
	cin >> t;
	FOR (z, 1, t) {
		cout << "Case #" << z << ": ";

		int n;
		cin >> n;

		FOR (i, 1, n) {
			cin >> a[i].s >> a[i].e >> a[i].l;
		}

		sort(a + 1, a + n + 1);

		ll res = 0;
		FOR (i, 1, n) {
			FOR (s, 0, 100000) {
				d[i][s] = d[i - 1][s];
				if (s >= a[i].s) {
					d[i][s] = max(d[i][s], d[i - 1][s - a[i].s] + max(0LL, a[i].e - (s - a[i].s) * 1ll * a[i].l));
				}

				if (i == n) res = max(res, d[i][s]);
			}
		}

		cout << res << endl;
	} 
    
	
    return 0;
}
