using namespace std;

typedef pair < int, int > pii;
typedef pair < ll, ll > pll;

int d[100005][27];
string s;

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(NULL);

    int t;
    cin >> t;

    FOR (z, 1, t) {
	    int n, q;
	    cin >> n >> q;
	    cin >> s;

	    s = '0' + s;

	    FOR (i, 1, n) {
	    	FOR (j, 0, 25) d[i][j] = d[i - 1][j];
	    	d[i][s[i] - 'A']++;
	    }

	    int res = 0;

	    FOR (i, 1, q) {
	    	int l, r;
	    	cin >> l >> r;

	    	int oddCnt = 0;
	    	FOR (j, 0, 25) {
	    		if ((d[r][j] - d[l - 1][j]) % 2) oddCnt++;
	    	}

	    	if (oddCnt <= 1) res++;
	    }

	    cout << "Case #" << z << ": " << res << endl;
	}
	
    return 0;
}
